# ==========================================

function Write-SAMLog {
    param([string]$Message, [string]$Level = "INFO")
    $Timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $Line = "$Timestamp [$Level] [Update] $Message"
    try { Add-Content -Path $LogFile -Value $Line -Encoding UTF8 -ErrorAction SilentlyContinue } catch {}
    if (-not $Silent) { Write-Host $Line -ForegroundColor Magenta }
}

function Get-FileHash256 {
    param([string]$FilePath)
    try {
        if (-not (Test-Path $FilePath)) { return "" }
        $SHA256 = [System.Security.Cryptography.SHA256]::Create()
        $Stream = [System.IO.File]::OpenRead($FilePath)
        $Hash   = [BitConverter]::ToString($SHA256.ComputeHash($Stream)) -replace '-'
        $Stream.Close()
        return $Hash.ToLower()
    } catch { return "" }
}

function Get-StringHash256 {
    param([string]$Content)
    try {
        $SHA256 = [System.Security.Cryptography.SHA256]::Create()
        $Bytes  = [System.Text.Encoding]::UTF8.GetBytes($Content)
        return ([BitConverter]::ToString($SHA256.ComputeHash($Bytes)) -replace '-').ToLower()
    } catch { return "" }
}

function Test-ScriptIntegrity {
    param([string]$FilePath, [string]$ExpectedHash = "")
    # ตรวจขั้นพื้นฐาน: ไฟล์มีอยู่และมีขนาดสมเหตุสมผล (> 1KB)
    if (-not (Test-Path $FilePath)) { return $false }
    $Size = (Get-Item $FilePath).Length
    if ($Size -lt 1024) {
        Write-SAMLog "File too small ($Size bytes): $FilePath" -Level "WARN"
        return $false
    }
    # ตรวจ syntax เบื้องต้น (ต้องมี param block หรือ function หรือ #)
    try {
        $Content = Get-Content $FilePath -Raw -ErrorAction Stop
        if ($Content -notmatch "(?i)(param\s*\(|function\s+\w|#.*SAM)") {
            Write-SAMLog "File failed syntax sanity check: $FilePath" -Level "WARN"
            return $false
        }
    } catch { return $false }
    # ถ้ามี expected hash ให้ตรวจด้วย
    if ($ExpectedHash -and $ExpectedHash.Length -eq 64) {
        $ActualHash = Get-FileHash256 $FilePath
        if ($ActualHash -ne $ExpectedHash.ToLower()) {
            Write-SAMLog "Hash mismatch: $FilePath (expected $ExpectedHash, got $ActualHash)" -Level "WARN"
            return $false
        }
    }
    return $true
}

function Download-SafeScript {
    param([string]$Url, [string]$OutPath, [string]$ExpectedHash = "", [string]$Label = "")
    $TmpPath = "$env:TEMP\SAM_tmp_$([System.IO.Path]::GetRandomFileName()).ps1"
    try {
        Invoke-WebRequest -Uri $Url -OutFile $TmpPath -UseBasicParsing -TimeoutSec 60 -ErrorAction Stop

        # Verify integrity
        if (-not (Test-ScriptIntegrity -FilePath $TmpPath -ExpectedHash $ExpectedHash)) {
            Write-SAMLog "Downloaded file failed integrity check: $Label" -Level "ERROR"
            Remove-Item $TmpPath -Force -ErrorAction SilentlyContinue
            return $false
        }

        # Atomic replace: backup → replace → remove backup
        $BackupPath = "$OutPath.bak"
        if (Test-Path $OutPath) {
            Copy-Item $OutPath $BackupPath -Force -ErrorAction SilentlyContinue
        }
        Move-Item $TmpPath $OutPath -Force -ErrorAction Stop
        Remove-Item $BackupPath -Force -ErrorAction SilentlyContinue

        Write-SAMLog "Updated successfully: $Label ($OutPath)"
        return $true
    } catch {
        Write-SAMLog "Download failed for $Label ($Url): $($_.Exception.Message)" -Level "ERROR"
        # Rollback ถ้ามี backup
        if (Test-Path "$OutPath.bak") {
            Copy-Item "$OutPath.bak" $OutPath -Force -ErrorAction SilentlyContinue
            Write-SAMLog "Rolled back to backup: $Label" -Level "WARN"
        }
        Remove-Item $TmpPath -Force -ErrorAction SilentlyContinue
        return $false
    }
}

# ==========================================
# SELF-HEALING: ตรวจไฟล์สำคัญก่อน
# ==========================================
Write-SAMLog "--- Self-Healing Check ---"

$FilesToCheck = @{
    $AgentScript     = @{ Url = $AgentUrl;      Label = "SAM_Agent" }
    $SchedulerScript = @{ Url = $SchedulerUrl;  Label = "SAM_Scheduler" }
    $UpdateScript    = @{ Url = $UpdateUrl;      Label = "SAM_Update" }
}

$HealingOccurred = $false
foreach ($FilePath in $FilesToCheck.Keys) {
    $Info = $FilesToCheck[$FilePath]
    if (-not (Test-ScriptIntegrity -FilePath $FilePath)) {
        Write-SAMLog "Corrupted/missing: $($Info.Label). Attempting self-heal..." -Level "WARN"
        $Healed = Download-SafeScript -Url $Info.Url -OutPath $FilePath -Label $Info.Label
        if ($Healed) {
            $HealingOccurred = $true
            Write-SAMLog "Self-healed: $($Info.Label)" -Level "INFO"
        } else {
            Write-SAMLog "Self-heal FAILED: $($Info.Label)" -Level "ERROR"
        }
    } else {
        Write-SAMLog "OK: $($Info.Label)"
    }
}

Write-SAMLog "--- Scheduled Tasks Health Check ---"
$TasksOK = $true
foreach ($TName in @($TASK_AGENT_SYNC, $TASK_C2_POLL, $TASK_FULL_SYNC, $TASK_UPDATE_CHECK, $TASK_LOG_ROTATE)) {
    $Task = Get-ScheduledTask -TaskName $TName -ErrorAction SilentlyContinue

        $TasksOK = $false

        Write-SAMLog "Task disabled: $TName → Enabling..." -Level "WARN"

        $TasksOK = $false
    }
}

if (-not $TasksOK) {

# ==========================================
# VERSION CHECK & UPDATE
# ==========================================
Write-SAMLog "--- Version Check ---"

# โหลด Update State (ป้องกันตรวจซ้ำในวันเดียวกัน)
$UpdateState = $null
if (Test-Path $UpdateStateFile) {
    try { $UpdateState = Get-Content $UpdateStateFile -Raw | ConvertFrom-Json } catch {}
}
$LastCheck = if ($UpdateState -and $UpdateState.LastCheck) { [datetime]$UpdateState.LastCheck } else { [datetime]::MinValue }
$HoursSince = ((Get-Date) - $LastCheck).TotalHours

if (-not $ForceCheck -and $HoursSince -lt 20) {
    Write-SAMLog "Update check skipped (last check $([math]::Round($HoursSince,1)) hrs ago). Use -ForceCheck to override."
    # ยังตรวจ self-healing ข้างบนแล้ว แต่ข้ามการดาวน์โหลด version ใหม่
} else {
    try {
        $VersionRaw = (Invoke-WebRequest -Uri $VersionUrl -UseBasicParsing -TimeoutSec 30 -ErrorAction Stop).Content.Trim()
        # Format: VERSION|hash_agent|hash_scheduler|hash_update
        $VerParts    = $VersionRaw -split '\|'
        $RemoteVer   = $VerParts[0].Trim()
        $RemoteHashA = if ($VerParts.Count -gt 1) { $VerParts[1].Trim() } else { "" }
        $RemoteHashS = if ($VerParts.Count -gt 2) { $VerParts[2].Trim() } else { "" }
        $RemoteHashU = if ($VerParts.Count -gt 3) { $VerParts[3].Trim() } else { "" }

        Write-SAMLog "Local: v$SAM_VERSION | Remote: v$RemoteVer"

        # เปรียบเทียบ version (simple semver string compare)
        $LocalParts  = $SAM_VERSION -split '\.' | ForEach-Object { [int]$_ }
        $RemoteParts = $RemoteVer -split '\.' | ForEach-Object { [int]$_ }
        $NeedUpdate  = $false

        for ($i = 0; $i -lt [math]::Max($LocalParts.Count, $RemoteParts.Count); $i++) {
            $L = if ($i -lt $LocalParts.Count)  { $LocalParts[$i] }  else { 0 }
            $R = if ($i -lt $RemoteParts.Count) { $RemoteParts[$i] } else { 0 }
            if ($R -gt $L) { $NeedUpdate = $true; break }
            if ($L -gt $R) { break }
        }

        if ($NeedUpdate) {
            Write-SAMLog "New version available: v$RemoteVer. Updating..."

            $UpdateResults = @{}

            # ลำดับการอัปเดต: Update ก่อน (เพื่อให้ตัวใหม่ทำงานต่อ), แล้ว Scheduler, แล้ว Agent
            $UpdateResults["SAM_Update"]     = Download-SafeScript -Url $UpdateUrl      -OutPath $UpdateScript    -ExpectedHash $RemoteHashU -Label "SAM_Update"
            $UpdateResults["SAM_Scheduler"]  = Download-SafeScript -Url $SchedulerUrl   -OutPath $SchedulerScript -ExpectedHash $RemoteHashS -Label "SAM_Scheduler"
            $UpdateResults["SAM_Agent"]      = Download-SafeScript -Url $AgentUrl       -OutPath $AgentScript     -ExpectedHash $RemoteHashA -Label "SAM_Agent"

            $AllOK = $UpdateResults.Values | Where-Object { $_ -eq $false } | Measure-Object | Select-Object -ExpandProperty Count
            if ($AllOK -eq 0) {
                Write-SAMLog "All files updated to v$RemoteVer. Re-installing tasks..."
                # Re-register tasks (อาจมีการเปลี่ยน argument หรือ schedule ใน version ใหม่)
                try {
                    Start-Process powershell.exe -ArgumentList "-WindowStyle Hidden -ExecutionPolicy Bypass -File `"$SchedulerScript`" -Setup" -WindowStyle Hidden -Wait
                    Write-SAMLog "Tasks re-registered post-update."
                } catch { Write-SAMLog "Task re-registration post-update failed." -Level "WARN" }
            } else {
                Write-SAMLog "Partial update — $AllOK file(s) failed. Will retry next cycle." -Level "WARN"
            }
        } else {
            Write-SAMLog "Already on latest version (v$SAM_VERSION)."

            # แม้ไม่มี version ใหม่ ให้ตรวจ hash ว่าไฟล์ถูกแก้ไขหรือเสียหรือเปล่า
            if ($RemoteHashA -or $RemoteHashS -or $RemoteHashU) {
                Write-SAMLog "Verifying file hashes against remote..."
                $HashChecks = @{
                    $AgentScript     = $RemoteHashA
                    $SchedulerScript = $RemoteHashS
                    $UpdateScript    = $RemoteHashU
                }
                foreach ($FP in $HashChecks.Keys) {
                    $ExpHash = $HashChecks[$FP]
                    if ($ExpHash -and $ExpHash.Length -eq 64) {
                        $ActHash = Get-FileHash256 $FP
                        if ($ActHash -ne $ExpHash.ToLower()) {
                            Write-SAMLog "Hash mismatch detected for $FP — restoring from remote." -Level "WARN"
                            # ดาวน์โหลดซ่อมจาก URL ที่ตรงกัน
                            $TargetUrl = switch ($FP) {
                                $AgentScript     { $AgentUrl }
                                $SchedulerScript { $SchedulerUrl }
                                $UpdateScript    { $UpdateUrl }
                            }
                            Download-SafeScript -Url $TargetUrl -OutPath $FP -ExpectedHash $ExpHash -Label (Split-Path $FP -Leaf) | Out-Null
                        }
                    }
                }
            }
        }

        # บันทึกเวลาตรวจล่าสุด
        try {
            @{ LastCheck = (Get-Date).ToString("o"); LastVersion = $RemoteVer } |
                ConvertTo-Json | Out-File $UpdateStateFile -Encoding UTF8
        } catch {}

    } catch {
        Write-SAMLog "Version check failed: $($_.Exception.Message)" -Level "ERROR"
    }
}
