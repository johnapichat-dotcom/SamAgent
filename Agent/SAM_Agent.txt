function Write-SAMLog {
    param([string]$Message, [string]$Level = "INFO")
    $Timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $Line = "$Timestamp [$Level] [Agent] $Message"
    try { Add-Content -Path $LogFile -Value $Line -Encoding UTF8 -ErrorAction SilentlyContinue } catch {}
    if (-not $Silent) { Write-Host $Line -ForegroundColor Cyan }
}

function Write-SAMChangeLog {
    # บันทึกเหตุการณ์เปลี่ยนแปลงลง Log แยก (เก็บไว้ถาวรกว่า 10 วัน = เก็บทุก event)
    param([string]$FieldName, [string]$OldValue, [string]$NewValue)
    $ChangeLogFile = "$WorkDir\SAM_ChangeLog.txt"
    $Timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $Line = "$Timestamp [CHANGE] $FieldName | OLD: $OldValue | NEW: $NewValue"
    try { Add-Content -Path $ChangeLogFile -Value $Line -Encoding UTF8 -ErrorAction SilentlyContinue } catch {}
    Write-SAMLog "Change detected: [$FieldName] $OldValue → $NewValue" -Level "CHANGE"
}

function Get-FieldHash {
    param([string]$Value)
    $MD5 = New-Object Security.Cryptography.MD5CryptoServiceProvider
    return [BitConverter]::ToString(
        $MD5.ComputeHash([Text.Encoding]::UTF8.GetBytes($Value))
    ) -replace '-'
}

# ==========================================
# C2 COMMAND PROCESSOR (function ใช้ร่วมกัน)
# ==========================================
function Invoke-C2Command {
    param([string]$RowId, [string]$RemoteCmd, [string]$SafeSerial)

    $Commands = $RemoteCmd -split "&&"
    foreach ($Cmd in $Commands) {
        $C = $Cmd.Trim()
        switch -Regex ($C) {
            "^SHUTDOWN$"    { Start-Process "shutdown.exe" -ArgumentList "/s /t 60 /c `"SAM Remote Command`"" -WindowStyle Hidden }
            "^RESTART$"     { Start-Process "shutdown.exe" -ArgumentList "/r /t 60 /c `"SAM Remote Command`"" -WindowStyle Hidden }
            "^CLEAR_TEMP$"  { Remove-Item "$env:TEMP\*" -Recurse -Force -ErrorAction SilentlyContinue }
            "^FORCE_SYNC$"  { Remove-Item $StateFile -Force -ErrorAction SilentlyContinue; Write-SAMLog "[C2] State cleared → FullSync next cycle" }
            "^RUN_TASK:(.+)$" {
                $Parts    = $Matches[1].Trim() -split '\|'
                $TaskFile = $Parts[0].Trim()
                $TaskArgs = if ($Parts.Count -gt 1) { ($Parts[1..($Parts.Count-1)] | ForEach-Object { "`"$_`"" }) -join " " } else { "" }
                $TaskUrl  = "$TaskBaseUrl$TaskFile"
                $TmpPath  = "$env:TEMP\$TaskFile"
                try {
                    Invoke-WebRequest -Uri $TaskUrl -OutFile $TmpPath -UseBasicParsing -ErrorAction Stop
                    $PSA = "-WindowStyle Hidden -ExecutionPolicy Bypass -Command `"& { try { & '$TmpPath' $TaskArgs } finally { Start-Sleep 2; Remove-Item '$TmpPath' -Force -ErrorAction SilentlyContinue } }`""
                    Start-Process powershell.exe -ArgumentList $PSA -WindowStyle Hidden
                    Write-SAMLog "[C2] Task dispatched: $TaskFile"
                } catch { Write-SAMLog "[C2] Task failed: $TaskFile — $($_.Exception.Message)" -Level "WARN" }
            }
            Default { Write-SAMLog "[C2] Unknown command: $C" -Level "WARN" }
        }
    }

    # Acknowledge → DONE
    try {
        $PatchUrl  = "https://xabvhfoosmqyrbkohmof.supabase.co/rest/v1/sam_devices?id=eq.$RowId"
        $PatchBody = @{ RemoteCommand = "DONE" } | ConvertTo-Json -Compress
        $PatchHdr  = @{ "apikey"=$SupabaseKey; "Authorization"="Bearer $SupabaseKey"; "Content-Type"="application/json"; "User-Agent"="SAM-Agent/v$SAM_VERSION" }
        Invoke-RestMethod -Uri $PatchUrl -Method Patch -Headers $PatchHdr -Body $PatchBody -TimeoutSec 10 | Out-Null
        Write-SAMLog "[C2] Acknowledged (DONE): $RowId"
    } catch { Write-SAMLog "[C2] Ack failed: $($_.Exception.Message)" -Level "WARN" }
}

        $SafeSerial = ""

        $RawSerial = if ($BIOS.SerialNumber) { $BIOS.SerialNumber.Trim() } else { "" }
        if ([string]::IsNullOrWhiteSpace($RawSerial) -or $RawSerial -match "(?i)To be filled|O\.E\.M\.|Default|None") {
            $UUID = (Get-CimInstance Win32_ComputerSystemProduct).UUID
            $SafeSerial = "HWID-" + ($UUID.Substring(0,18))
        } else { $SafeSerial = $RawSerial }

        $PollUrl = "https://xabvhfoosmqyrbkohmof.supabase.co/rest/v1/sam_devices?SerialNumber=eq.$([uri]::EscapeDataString($SafeSerial))&select=id,RemoteCommand"
        $PollHeaders = @{
            "apikey"        = $SupabaseKey
            "Authorization" = "Bearer $SupabaseKey"
            "User-Agent"    = "SAM-Agent/v$SAM_VERSION"
        }
        $PollResponse = Invoke-RestMethod -Uri $PollUrl -Method Get -Headers $PollHeaders -TimeoutSec 15 -ErrorAction Stop

        if ($PollResponse -and $PollResponse.Count -gt 0) {
            $Row = $PollResponse[0]
            if ($Row.RemoteCommand -and $Row.RemoteCommand -notin @("","NONE","DONE")) {
                Write-SAMLog "C2 Command received: $($Row.RemoteCommand)"
                Invoke-C2Command -RowId $Row.id -RemoteCmd $Row.RemoteCommand -SafeSerial $SafeSerial
            } else {
                Write-SAMLog "C2 Poll: No pending command."
            }
        }
    } catch {
        Write-SAMLog "C2 Poll failed: $($_.Exception.Message)" -Level "WARN"
    }
    exit 0
}

# Antivirus — รองรับ Defender, Trend Micro Apex One, และ AV อื่นๆ
function Get-AVVersion {
    param([string]$AVName)
    try {
        # Windows Defender
        if ($AVName -match "Defender|Microsoft") {
            $mpStatus = Get-MpComputerStatus -ErrorAction SilentlyContinue
            if ($mpStatus -and $mpStatus.AMProductVersion) { return $mpStatus.AMProductVersion }
        }
        # Trend Micro Apex One / OfficeScan
        if ($AVName -match "Trend Micro|Apex One|OfficeScan|PC-cillin") {
            $tmExeList = @(
                "$env:ProgramFiles\Trend Micro\Security Agent\Ntrtscan.exe",
                "${env:ProgramFiles(x86)}\Trend Micro\Security Agent\Ntrtscan.exe",
                "$env:ProgramFiles\Trend Micro\Apex One Security Agent\Ntrtscan.exe",
                "${env:ProgramFiles(x86)}\Trend Micro\OfficeScan Client\Ntrtscan.exe"
            )
            $tmExe = $tmExeList | Where-Object { Test-Path $_ } | Select-Object -First 1
            if ($tmExe) {
                $rawVer = (Get-Item $tmExe -ErrorAction SilentlyContinue).VersionInfo.FileVersion
                if ($rawVer) {
                    return ($rawVer -replace '^(\d+\.\d+)\.0\.(\d+)$','$1.$2')
                }
            }
            foreach ($p in @("HKLM:\SOFTWARE\TrendMicro\PC-cillinNTCorp\CurrentVersion","HKLM:\SOFTWARE\WOW6432Node\TrendMicro\PC-cillinNTCorp\CurrentVersion")) {
                if (Test-Path $p) {
                    $reg = Get-ItemProperty $p -ErrorAction SilentlyContinue
                    $ver = if ($reg.ProgramVer) { $reg.ProgramVer } elseif ($reg.AgentVersion) { $reg.AgentVersion } else { $null }
                    if ($ver) { return $ver }
                }
            }
        }
        # Generic: ค้นหาใน Uninstall registry
        $searchName = $AVName -replace " Antivirus| Security| Endpoint| Agent",""
        $swPaths = @("HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\*",
                     "HKLM:\SOFTWARE\WOW6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*")
        $found = Get-ItemProperty $swPaths -ErrorAction SilentlyContinue |
            Where-Object { $_.DisplayName -and $_.DisplayName -match [regex]::Escape($searchName) } |
            Select-Object -First 1
        if ($found -and $found.DisplayVersion) { return $found.DisplayVersion }
    } catch {}
    return "Unknown"
}

Write-SAMLog "Analyzing changes (Delta Engine)..."

# โหลด State เดิม
$LastState     = $null
$LastFieldHash = @{}
$LastFullSync  = $null

if (Test-Path $StateFile) {
    try {
        $LastState    = Get-Content $StateFile -Raw | ConvertFrom-Json
        $LastFullSync = if ($LastState.LastFullSync) { [datetime]$LastState.LastFullSync } else { $null }
        # ดึง field hashes เดิม (ถ้ามี)
        if ($LastState.FieldHashes) {
            $LastState.FieldHashes.PSObject.Properties | ForEach-Object {
                $LastFieldHash[$_.Name] = $_.Value
            }
        }
    } catch { Write-SAMLog "State file corrupt, will force FullSync." -Level "WARN" }

# ตรวจสอบว่าถึงเวลา Force FullSync หรือยัง
$DaysSinceFullSync = if ($LastFullSync) { ((Get-Date) - $LastFullSync).TotalDays } else { 999 }
$NeedFullSync      = $ForceFullSync -or ($DaysSinceFullSync -ge $FULL_SYNC_INTERVAL_DAYS) -or ($null -eq $LastState)

# Hash แต่ละ field เพื่อหาที่เปลี่ยนแปลง
$CurrentFieldHash = @{}
$ChangedFields    = @()
$DeltaPayload     = [ordered]@{
    SerialNumber = $SafeSerial
    ComputerName = $env:COMPUTERNAME
    updated_at   = $CurrentData.updated_at
    LastSyncTH   = $CurrentData.LastSyncTH   # ส่งทุก sync type เสมอ — สัญญาณชีพ
    AgentVersion = $CurrentData.AgentVersion  # ส่งทุกครั้งเพื่อยืนยัน version
}

foreach ($Key in $CurrentData.Keys) {
    $Val  = if ($null -ne $CurrentData[$Key]) { $CurrentData[$Key].ToString() } else { "" }
    $Hash = Get-FieldHash $Val
    $CurrentFieldHash[$Key] = $Hash

    if (-not $NeedFullSync) {
        $OldHash = $LastFieldHash[$Key]
        if ($OldHash -ne $Hash) {
            $ChangedFields += $Key
            $DeltaPayload[$Key] = $CurrentData[$Key]
            # บันทึก Change Log
            $OldVal = if ($LastState -and $LastState.FieldHashes -and $LastState.FieldData -and $LastState.FieldData.$Key) { $LastState.FieldData.$Key } else { "(unknown)" }
            Write-SAMChangeLog -FieldName $Key -OldValue $OldVal -NewValue $Val
        }
    }
}

# กำหนดประเภทการส่ง

    $DeltaPayload["ChangedFields"] = ($ChangedFields -join ",")
    $PayloadData = $DeltaPayload
    Write-SAMLog "Sync type: DeltaSync — $($ChangedFields.Count) field(s) changed: $($ChangedFields -join ', ')"
} else {
    # ไม่มีอะไรเปลี่ยน → ส่ง Heartbeat เบาๆ
    $SyncType = "Heartbeat"
    $DeltaPayload["Type"] = "Heartbeat"

$Headers = @{
    "apikey"        = $SupabaseKey
    "Authorization" = "Bearer $SupabaseKey"
    "Content-Type"  = "application/json; charset=utf-8"
    "Prefer"        = "resolution=merge-duplicates, return=representation"
    "User-Agent"    = "SAM-Agent/v$SAM_VERSION"
}

$JsonPayload = ($PayloadData | ConvertTo-Json -Depth 5 -Compress) -replace '\\u0000', ''
$Utf8Bytes   = [System.Text.Encoding]::UTF8.GetBytes($JsonPayload)
$MaxRetries  = 6; $Attempt = 0; $IsSuccess = $false

while ($Attempt -lt $MaxRetries -and -not $IsSuccess) {
    $Attempt++
    try {
        $Timer = [System.Diagnostics.Stopwatch]::StartNew()
        $Response = Invoke-RestMethod -Uri $SupabaseUrl -Method Post -Headers $Headers -Body $Utf8Bytes -TimeoutSec 30 -ErrorAction Stop
        $Timer.Stop()
        $IsSuccess = $true

        $ElapsedS = [math]::Round($Timer.Elapsed.TotalSeconds, 2)
        $SizeKB   = [math]::Round($Utf8Bytes.Length / 1KB, 2)
        Write-SAMLog "Supabase sync OK. Type:$SyncType | Attempt:$Attempt | $SizeKB KB | $ElapsedS s"

        # อัปเดต State File
        $NewState = @{
            Hash         = (Get-FieldHash $JsonPayload)
            LastFullSync = if ($SyncType -eq "FullSync") { (Get-Date).ToString("o") } else { if ($LastFullSync) { $LastFullSync.ToString("o") } else { "" } }
            LastWriteTime = (Get-Date).ToString("o")
            FieldHashes  = $CurrentFieldHash
            # เก็บค่าปัจจุบันไว้เปรียบเทียบรอบหน้า (เฉพาะ field สำคัญ ไม่เก็บ SW list ยาว)
            FieldData    = @{
                OSVersion = $CurrentData.OSVersion; RAM = $CurrentData.RAM; IPAddress = $CurrentData.IPAddress
                DriveSpace = $CurrentData.DriveSpace; DiskHealth = $CurrentData.DiskHealth
                Antivirus1 = $CurrentData.Antivirus1; WinUpdate = $CurrentData.WinUpdate
                BranchName = $CurrentData.BranchName; DeviceType = $CurrentData.DeviceType
            }
        }
        try { $NewState | ConvertTo-Json -Depth 5 | Out-File $StateFile -Encoding UTF8 } catch {}

        # C2 Command Processing
        $RemoteCmd = if ($Response -is [array]) { $Response[0].RemoteCommand } else { $Response.RemoteCommand }
        $RowId     = if ($Response -is [array]) { $Response[0].id } else { $Response.id }
        if ($RemoteCmd -and $RemoteCmd -notin @("","NONE","DONE")) {
            Write-SAMLog "C2 Command: $RemoteCmd"
            Invoke-C2Command -RowId $RowId -RemoteCmd $RemoteCmd -SafeSerial $SafeSerial
        }

    } catch {
        if ($null -ne $Timer) { $Timer.Stop() }
        $ErrMsg = if ($_.ErrorDetails) { $_.ErrorDetails.Message } else { $_.Exception.Message }
        $RetryWait = Get-Random -Minimum 60 -Maximum 300
        Write-SAMLog "Sync failed (Attempt $Attempt/$MaxRetries): $ErrMsg. Retry in $RetryWait s." -Level "WARN"
        if ($Attempt -lt $MaxRetries) { Start-Sleep -Seconds $RetryWait }
    }
}

if (-not $IsSuccess) {
    Write-SAMLog "All $MaxRetries sync attempts failed." -Level "ERROR"
}

function Build-HTMLReport {
    $CleanBranch = $BranchID -replace '[\\/:*?"<>|]', '-'
    $ReportPath  = "$WorkDir\$($BU)_$($CleanBranch)_$($env:COMPUTERNAME)_$(Get-Date -Format 'yyyy-MM-dd').html"

    # ลบ report เก่าของเครื่องนี้
    Get-ChildItem $WorkDir -Filter "$($BU)_$($CleanBranch)_$($env:COMPUTERNAME)*.html" |
        Where-Object { $_.FullName -ne $ReportPath } | Remove-Item -Force -ErrorAction SilentlyContinue

    $CSS = "<style>body{font-family:'Leelawadee UI',Tahoma,sans-serif;margin:20px}table{border-collapse:collapse;width:95%;margin:auto auto 25px;box-shadow:0 2px 3px rgba(0,0,0,.1)}th,td{border:1px solid #ddd;text-align:left;padding:8px}th{background-color:#f2f2f2}tr:nth-child(even){background-color:#f9f9f9}h1,h2{text-align:center;color:#333}h1{border-bottom:2px solid #eee;padding-bottom:10px}.delta{color:#0056b3;font-weight:bold}</style>"
    $H  = "<!DOCTYPE html><html><head><meta charset='UTF-8'>$CSS</head><body>"
    $H += "<h1>System Audit — $env:COMPUTERNAME<br><span style='font-size:0.6em;color:#666;'>BU:$BU | Branch:$BranchID | Prov:$Province | Agent:v$SAM_VERSION | Sync:$SyncType</span></h1>"

    $H += "<h2>System Information</h2><table>"
    foreach ($K in $CurrentData.Keys | Where-Object { $_ -notin @("updated_at","LastSyncTH","Type","ChangedFields","AllSoftware") }) {
        $MarkClass = if ($ChangedFields -contains $K) { " class='delta'" } else { "" }
        $H += "<tr><td$MarkClass>$K</td><td$MarkClass>$($CurrentData[$K])</td></tr>"
    }
    $H += "</table>"

    if ($AllSW.Count -gt 0) {
        $H += "<h2>Installed Applications</h2><table><tr><th>Name</th><th>Version</th><th>Install Date</th></tr>"
        foreach ($App in $AllSW) {
            $H += "<tr><td>$($App.DisplayName)</td><td>$($App.DisplayVersion)</td><td>$($App.InstallDate)</td></tr>"
        }
        $H += "</table>"
    }

    $H += "</body></html>"
    try { [System.IO.File]::WriteAllText($ReportPath, $H, [System.Text.Encoding]::UTF8) } catch {}
    if (-not $Silent) { try { Invoke-Item $ReportPath } catch {} }
    Write-SAMLog "HTML report saved: $ReportPath"
}
